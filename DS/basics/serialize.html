Serialize in Java


package HackerRank;

import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.io.Serializable;

class Emp implements Serializable {
	String empName;
	int age;
	int salary;

	public Emp(String empName, int age, int salary) {
		this.empName = empName;
		this.age = age;
		this.salary = salary;
	}
}

class Ex {
	private static Emp deserialise() {
		Emp e = null;
		try {
			FileInputStream fileIn = new FileInputStream("emp.txt");
			ObjectInputStream in = new ObjectInputStream(fileIn);

			e = (Emp) in.readObject();
		} catch (Exception e1) {
			System.out.println(e1);
		}
		return e;
	}

	private static void serialiseObj(Emp emp) {
		try {
			FileOutputStream fileOut = new FileOutputStream("emp.txt");
			ObjectOutputStream out = new ObjectOutputStream(fileOut);
			out.writeObject(emp);
			out.flush();
			out.close();
			System.out.println("emp Obj Serializing Done");
		} catch (Exception e) {
			System.out.println(e);
		}
	}
	
	public static void main(String[] args) {
		Emp emp = new Emp("John Wick", 30, 2500000);
		serialiseObj(emp);

		Emp desObj = deserialise();
		System.out.println(desObj.empName);
		System.out.println(desObj.age);
		System.out.println(desObj.salary);
	}
}
